## Installation

```bash
npm install bm-signature
```

## Use

```js
import {
      printToDOM,
      printToConsole,
      printToDOMasComment,
    } from "bm-signature"

printToConsole(): void;
printToDOM(e: HTMLPreElement): void;
printToDOMasComment(e: HTMLElement): void;
```
